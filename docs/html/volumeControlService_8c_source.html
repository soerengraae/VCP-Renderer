119/* GATT write handler for Volume Control Point (0x2B7E) */
120bt_gatt_attr_write_func_t writeVolumeControlPoint(struct bt_conn *conn, const struct bt_gatt_attr *attr, const void *buf, uint16_t len, uint16_t offset, uint8_t flags)
121{
122 if (offset != 0 || (len != 2 && len != 3)) {
123 LOG_WRN("Invalid attribute length: %d\n", len);
124 return BT_GATT_ERR(BT_ATT_ERR_INVALID_ATTRIBUTE_LEN);
125 }
126
127 uint8_t opcode = ((uint8_t *)buf)[0];
128 uint8_t operand = ((uint8_t *)buf)[1];
129
130 if (operand != vcsState.changeCounter) {
131 LOG_WRN("Invalid Change Counter: %d (expected %d)\n", operand, vcsState.changeCounter);
132 return BT_GATT_ERR(ERR_INVALID_CHANGE_COUNTER);
133 }
134
135 switch (opcode) {
136 case VOLUME_DOWN:
137 LOG_DBG("Opcode: VOLUME_DOWN\n");
138 volumeDown(&vcsState.volumeSetting);
139 break;
140 case VOLUME_UP:
141 LOG_DBG("Opcode: VOLUME_UP\n");
142 volumeUp(&vcsState.volumeSetting);
143 break;
144 case VOLUME_DOWN_UNMUTE:
145 LOG_DBG("Opcode: VOLUME_DOWN_UNMUTE\n");
146 volumeDown(&vcsState.volumeSetting);
147 volumeUnmute();
148 break;
149 case VOLUME_UP_UNMUTE:
150 LOG_DBG("Opcode: VOLUME_UP_UNMUTE\n");
151 volumeUp(&vcsState.volumeSetting);
152 volumeUnmute();
153 break;
154 case VOLUME_SET_ABSOLUTE:
155 LOG_DBG("Opcode: VOLUME_SET_ABSOLUTE\n");
156 if (len != 3) {
157 LOG_WRN("Invalid attribute length for VOLUME_SET_ABSOLUTE: %d\n", len);
158 return BT_GATT_ERR(BT_ATT_ERR_INVALID_ATTRIBUTE_LEN);
159 }
160 operand = ((uint8_t *)buf)[2]; // Casting it to uint8_t makes sure it's in 0-255 range
161 volume_set(&vcsState.volumeSetting, operand);
162 break;
163 case VOLUME_UNMUTE:
164 LOG_DBG("Opcode: VOLUME_UNMUTE\n");
165 volumeUnmute();
166 break;
167 case VOLUME_MUTE:
168 LOG_DBG("Opcode: VOLUME_MUTE\n");
169 volumeMute();
170 break;
171
172 default:
173 LOG_WRN("Invalid Opcode: %d\n", opcode);
174 return BT_GATT_ERR(ERR_INVALID_OPCODE);
175 }
176
177 switch (opcode) {
178 case VOLUME_DOWN:
179 case VOLUME_UP:
180 case VOLUME_DOWN_UNMUTE:
181 case VOLUME_UP_UNMUTE:
182 case VOLUME_SET_ABSOLUTE:
183 volumeFlagsSet((0x01 << 0), conn); // Volume changed - set Volume_Setting_Persisted flag
184 break;
185
186 default:
187 break;
188 }
189
190 changeCounterIncrement(conn);
191
192 return len; // Success - return number of bytes processed
193}
